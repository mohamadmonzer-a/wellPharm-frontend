async function loadProducts() { try { const res = await fetch('http://localhost:3000/api/medicines'); if (!res.ok) throw new Error('Failed to fetch'); const products = await res.json(); const productsSection = document.querySelector('.products'); productsSection.innerHTML = ''; // Clear existing static cards products.forEach(product => { const card = document.createElement('div'); card.className = 'product-card'; card.innerHTML = `

${product.name}

$${product.price.toFixed(2)}

`; productsSection.appendChild(card); }); } catch (error) { console.error(error); } } function addToCart(productName) { alert(`${productName} added to cart!`); } window.onload = loadProducts;
